\section{Modelo Linear (ARX)}

\begin{frame}
  \begin{align}
    % Definição das variáveis
    y(k) & =
    \begin{bmatrix}
      \text{purity}_{H_2}(k)    \\
      \text{H}_2/\text{CO}(k)   \\
      \text{purity}_{CO_2}(k)   \\
      \text{recovery}_{CO_2}(k) \\
      \text{productivity}(k)
    \end{bmatrix},
         &
    u(k) & =
    \begin{bmatrix}
      t_{\text{feed}}(k)  \\
      t_{\text{rinse}}(k) \\
      t_{\text{blow}}(k)  \\
      t_{\text{purge}}(k)
    \end{bmatrix}
  \end{align}

  \begin{align}
    \varphi(k-1) & =
    \begin{bmatrix}
      y(k-1) \\
      u(k-1)
    \end{bmatrix}
    \\[1em]
    y(k)         & = \varphi(k-1)^\top \theta
  \end{align}
\end{frame}
