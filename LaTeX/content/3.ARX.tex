\section{Modelo Linear (ARX)}

\begin{frame}
  \begin{equation}
    % Definição das variáveis
    y(k) =
    \begin{bmatrix}
      \text{purity}_{\ce{H2}}(k)    \\
      \ce{H2}/\ce{CO}(k)            \\
      \text{purity}_{\ce{CO2}}(k)   \\
      \text{recovery}_{\ce{CO2}}(k) \\
      \text{productivity}(k)
    \end{bmatrix},
    \qquad
    u(k) =
    \begin{bmatrix}
      t_{\text{feed}}(k)  \\
      t_{\text{rinse}}(k) \\
      t_{\text{blow}}(k)  \\
      t_{\text{purge}}(k)
    \end{bmatrix}
  \end{equation}

  \begin{align}
    \varphi(k) & =
    \begin{bmatrix}
      y(k-1) \\
      u(k-1) \\
      u(k)
    \end{bmatrix}
    \\[1em]
    y(k)       & = \varphi(k)^\top \theta
  \end{align}
\end{frame}
