\section{Modelo não Linear (NARX)}

\begin{frame}
  \begin{align}
    \varphi(k) & =
    \begin{bmatrix}
      y(k-1)                \\[0.3em]
      u(k-1)                \\[0.3em]
      u(k)                  \\[0.5em]
      y(k-1)^2              \\[0.3em]
      u(k-1)^2              \\[0.3em]
      u(k)^2                \\[0.5em]
      y(k-1) \otimes u(k-1) \\[0.3em]
      y(k-1) \otimes u(k)   \\
    \end{bmatrix}
    \\[1em]
    y(k)       & = \varphi(k)^\top \theta
  \end{align}
\end{frame}
